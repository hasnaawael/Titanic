      ###loop over sex and pclass to calculate age
      for dataset in combine:
          ###Get in trian data and ignore null values
          for i in range(0,2):### 0, 1 === Gender
              for j in range(0,3): ### 0, 1, 2 === pclass
                  guess_df = dataset[(dataset['Sex'] == i)&  
                                    (dataset['Pclass']==j+1)]['Age'].dropna()
                  
                  age_guess = guess_df.median()### get this values
                  guess_ages[i,j] = int (age_guess/0.5+0.5)*0.5###convert float to nearest .5 age
                  
          ### loop in test data to fill null values       
          for i in range(0,2):
              for j in range(0,3):
                  dataset.loc[ (dataset.Age.isnull())&(dataset.Sex ==i)&(dataset.Pclass==j+1), 'Age']=guess_ages[i,j]
          dataset['Age'] = dataset['Age'].astype(int)
